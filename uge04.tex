\begin{opgavebreak}[4.6]
\begin{solutionsec}
    \item Bemærk at
    %
    \begin{equation*}
        \abs{p_j(x) - p_j(y)}^2
            = \abs{x_j - y_j}^2
            \leq \sum_{j=1}^d \abs{x_j - y_j}^2.
    \end{equation*}

    \item For $\phi_1$ og $\phi_2$, benyt trekantsuligheden. For $\phi_3$ og $\phi_4$, vis først at funktionen $x \mapsto \abs{x}$ er kontinuert (husk den omvendte trekantsulighed), og benyt eksempelvis at
    %
    \begin{equation*}
        x \meet y
            = \frac{x+y-\abs{x-y}}{2}.
    \end{equation*}
\end{solutionsec}
\end{opgavebreak}
